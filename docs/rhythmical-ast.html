function nested(fn, getChildren) {
  return (node) => {
    node = fn(node, true); // mutate before
    const children = getChildren(node);
    if (children?.length) {
      return children.map(nested(fn, getChildren));
    }
    return fn(node, false); // mutate after
  };
}
