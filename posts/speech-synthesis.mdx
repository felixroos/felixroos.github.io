const lang =

  ({ lang }) =>
    lang === 'de-DE';

function speak(words, voice) {
  const synth = window.speechSynthesis;
  utterance = new SpeechSynthesisUtterance(words);
  if (typeof voice === 'number') {
    synth.getVoices()[voice % synth.getVoices().length];
  } else if (typeof voice === 'object') {
    utterance.voice = voice;
  } else if (typeof voice === 'function') {
    utterance.voice = voice(synth.getVoices());
  }
  speechSynthesis.speak(utterance);
}
